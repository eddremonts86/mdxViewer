export const CodigoEjecutable = ({ codigo, lenguaje = 'javascript' }) => {
  const [resultado, setResultado] = useState('');
  const [ejecutando, setEjecutando] = useState(false);

  const ejecutarCodigo = () => {
    setEjecutando(true);
    try {
      // Simulación de ejecución de código
      setTimeout(() => {
        setResultado(`Resultado: ${eval(codigo)}`);
        setEjecutando(false);
      }, 1000);
    } catch (error) {
      setResultado(`Error: ${error.message}`);
      setEjecutando(false);
    }
  };

  return (
    <div style={{ margin: '20px 0' }}>
      <pre style={{
        backgroundColor: '#f8f9fa',
        padding: '15px',
        borderRadius: '5px',
        border: '1px solid #e9ecef',
        overflow: 'auto'
      }}>
        <code>{codigo}</code>
      </pre>
      <button
        onClick={ejecutarCodigo}
        disabled={ejecutando}
        style={{
          backgroundColor: '#28a745',
          color: 'white',
          border: 'none',
          padding: '8px 16px',
          borderRadius: '4px',
          cursor: ejecutando ? 'wait' : 'pointer',
          marginTop: '10px'
        }}
      >
        {ejecutando ? 'Ejecutando...' : 'Ejecutar Código'}
      </button>
      {resultado && (
        <div style={{
          marginTop: '10px',
          padding: '10px',
          backgroundColor: '#e9ecef',
          borderRadius: '4px',
          fontFamily: 'monospace'
        }}>
          {resultado}
        </div>
      )}
    </div>
  );
};

```javascript
import React, { useState, useEffect } from 'react';

const ComponenteEjemplo = ({ datos }) => {
  const [estado, setEstado] = useState(null);
  const [cargando, setCargando] = useState(true);

  useEffect(() => {
    const cargarDatos = async () => {
      try {
        setCargando(true);
        const respuesta = await fetch('/api/datos');
        const resultado = await respuesta.json();
        setEstado(resultado);
      } catch (error) {
        console.error('Error cargando datos:', error);
      } finally {
        setCargando(false);
      }
    };

    cargarDatos();
  }, []);

  if (cargando) {
    return <div>Cargando...</div>;
  }

  return (
    <div className="componente">
      <h2>Datos Cargados</h2>
      {estado && (
        <ul>
          {estado.map(item => (
            <li key={item.id}>
              {item.nombre}: {item.valor}
            </li>
          ))}
        </ul>
      )}
    </div>
  );
};

export default ComponenteEjemplo;
```

```typescript
interface DatosUsuario {
  id: number;
  nombre: string;
  email: string;
  activo: boolean;
}

type EstadoApp = 'cargando' | 'exitoso' | 'error';

class GestorUsuarios {
  private usuarios: DatosUsuario[] = [];
  private estado: EstadoApp = 'cargando';

  constructor(private apiUrl: string) {}

  async cargarUsuarios(): Promise<void> {
    try {
      this.estado = 'cargando';
      const respuesta = await fetch(`${this.apiUrl}/usuarios`);

      if (!respuesta.ok) {
        throw new Error(`Error HTTP: ${respuesta.status}`);
      }

      this.usuarios = await respuesta.json();
      this.estado = 'exitoso';
    } catch (error) {
      console.error('Error cargando usuarios:', error);
      this.estado = 'error';
    }
  }

  obtenerUsuariosActivos(): DatosUsuario[] {
    return this.usuarios.filter(usuario => usuario.activo);
  }

  buscarPorEmail(email: string): DatosUsuario | undefined {
    return this.usuarios.find(usuario => usuario.email === email);
  }
}
```

#### Python

```python
from typing import List, Dict, Optional
import asyncio
import aiohttp
from dataclasses import dataclass
from datetime import datetime

@dataclass
class Usuario:
    id: int
    nombre: str
    email: str
    activo: bool = True
    fecha_creacion: datetime = None

    def __post_init__(self):
        if self.fecha_creacion is None:
            self.fecha_creacion = datetime.now()

class GestorUsuariosAsync:
    def __init__(self, api_url: str):
        self.api_url = api_url
        self.usuarios: List[Usuario] = []

    async def cargar_usuarios(self) -> None:
        """Carga usuarios de la API de forma asíncrona."""
        async with aiohttp.ClientSession() as session:
            try:
                async with session.get(f"{self.api_url}/usuarios") as response:
                    response.raise_for_status()
                    datos = await response.json()

                    self.usuarios = [
                        Usuario(**usuario_data)
                        for usuario_data in datos
                    ]

            except aiohttp.ClientError as e:
                print(f"Error cargando usuarios: {e}")
                raise

    def filtrar_activos(self) -> List[Usuario]:
        """Retorna solo usuarios activos."""
        return [u for u in self.usuarios if u.activo]

    def buscar_por_email(self, email: str) -> Optional[Usuario]:
        """Busca un usuario por email."""
        return next(
            (u for u in self.usuarios if u.email == email),
            None
        )

    async def crear_usuario(self, datos: Dict) -> Usuario:
        """Crea un nuevo usuario."""
        async with aiohttp.ClientSession() as session:
            async with session.post(
                f"{self.api_url}/usuarios",
                json=datos
            ) as response:
                response.raise_for_status()
                usuario_data = await response.json()
                nuevo_usuario = Usuario(**usuario_data)
                self.usuarios.append(nuevo_usuario)
                return nuevo_usuario

# Ejemplo de uso
async def main():
    gestor = GestorUsuariosAsync("https://api.ejemplo.com")
    await gestor.cargar_usuarios()

    usuarios_activos = gestor.filtrar_activos()
    print(f"Usuarios activos: {len(usuarios_activos)}")

    # Crear nuevo usuario
    nuevo_usuario = await gestor.crear_usuario({
        "nombre": "Juan Pérez",
        "email": "juan@ejemplo.com"
    })
    print(f"Usuario creado: {nuevo_usuario.nombre}")

if __name__ == "__main__":
    asyncio.run(main())
```
